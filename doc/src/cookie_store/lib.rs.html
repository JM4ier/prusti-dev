#[derive(Debug, Fail)]
#[fail(display = "IDNA errors: {:#?}", _0)]
pub struct IdnaErrors(idna::uts46::Errors);

impl From<idna::uts46::Errors> for IdnaErrors {
    fn from(e: idna::uts46::Errors) -> Self {
        IdnaErrors(e)
    }
}
